// ===== Chart Theme Bridge =====
function getThemeColors() {
  var styles = getComputedStyle(document.documentElement);
  return {
    textMuted: styles.getPropertyValue('--text-muted').trim(),
    border: styles.getPropertyValue('--border').trim(),
    surface2: styles.getPropertyValue('--surface2').trim(),
    text: styles.getPropertyValue('--text').trim()
  };
}

function applyChartTheme() {
  var c = getThemeColors();
  Chart.defaults.color = c.textMuted;
  Chart.defaults.borderColor = c.border;
  [window.langChart, window.debtChart].forEach(function(chart) {
    if (!chart) return;
    chart.options.plugins.tooltip = chart.options.plugins.tooltip || {};
    chart.options.plugins.tooltip.backgroundColor = c.surface2;
    chart.options.plugins.tooltip.bodyColor = c.text;
    chart.update('none');
  });
}

// ===== Chart Init =====
var initColors = getThemeColors();
Chart.defaults.color = initColors.textMuted;
Chart.defaults.borderColor = initColors.border;

var langCfg = {{ lang_chart_json | safe }};
if (langCfg && langCfg.type) {
  langCfg.options = langCfg.options || {};
  langCfg.options.plugins = langCfg.options.plugins || {};
  langCfg.options.plugins.tooltip = { backgroundColor: initColors.surface2, bodyColor: initColors.text };
  window.langChart = new Chart(document.getElementById('langChart'), langCfg);
}

var debtCfg = {{ debt_chart_json | safe }};
if (debtCfg && debtCfg.type) {
  debtCfg.options = debtCfg.options || {};

  debtCfg.options.plugins.tooltip = { backgroundColor: initColors.surface2, bodyColor: initColors.text };

}

**D. Theme toggle JS** — Add AFTER the chart init block (still inside the bottom `<script>` tag):

```javascript
// ===== Theme Toggle =====
var THEME_STATES = [
  { key: 'dark',   icon: '\u{1F319}', next: 'Switch to Light mode' },
  { key: 'light',  icon: '\u2600',    next: 'Switch to System mode' },
  { key: 'system', icon: '\u{1F5A5}', next: 'Switch to Dark mode' }
];

function resolveTheme(key) {
  if (key === 'light') return 'light';
  if (key === 'dark') return null;  // null = remove attribute = dark default
  // system: detect OS preference
  try {
    if (window.matchMedia('(prefers-color-scheme: light)').matches) return 'light';
  } catch(e) {}
  return null;
}

function applyTheme(key) {
  var resolved = resolveTheme(key);
  if (resolved === 'light') {
    document.documentElement.setAttribute('data-theme', 'light');
  } else {
    document.documentElement.removeAttribute('data-theme');
  }
}

function updateToggleButton(key) {
  var btn = document.getElementById('themeToggle');
  if (!btn) return;
  var idx = THEME_STATES.findIndex(function(s) { return s.key === key; });
  if (idx === -1) idx = 0;
  btn.textContent = THEME_STATES[idx].icon;
  btn.title = THEME_STATES[idx].next;
  btn.setAttribute('aria-label', THEME_STATES[idx].next);
}

function toggleTheme() {
  var current;
  try { current = localStorage.getItem('gitopsy-theme') || 'dark'; } catch(e) { current = 'dark'; }
  var idx = THEME_STATES.findIndex(function(s) { return s.key === current; });
  var nextIdx = (idx + 1) % THEME_STATES.length;
  var nextKey = THEME_STATES[nextIdx].key;
  try { localStorage.setItem('gitopsy-theme', nextKey); } catch(e) {}
  applyTheme(nextKey);
  updateToggleButton(nextKey);
  applyChartTheme();
}

// Initialize toggle button state on load
(function() {
  var current;
  try { current = localStorage.getItem('gitopsy-theme'); } catch(e) { current = null; }
  if (!current) current = 'dark';
  updateToggleButton(current);
})();
```

```javascript
// ===== Score Animation (VIS-03) =====
document.addEventListener('DOMContentLoaded', function() {
  var el = document.querySelector('.score-overall-badge');
  if (!el) return;
  var finalScore = parseInt(el.dataset.score, 10);
  if (isNaN(finalScore)) return;
  var finalGrade = el.textContent.trim();
  var duration = 800;
  var start = null;

  function scoreToGrade(s) {
    if (s >= 90) return 'A';
    if (s >= 80) return 'B';
    if (s >= 65) return 'C';
    if (s >= 50) return 'D';
    return 'F';
  }

  function easeOut(t) { return 1 - (1 - t) * (1 - t); }

  function frame(ts) {
    if (!start) start = ts;
    var progress = Math.min((ts - start) / duration, 1);
    var easedProgress = easeOut(progress);
    var currentScore = Math.round(easedProgress * finalScore);
    el.textContent = progress < 1 ? scoreToGrade(currentScore) : finalGrade;
    if (progress < 1) requestAnimationFrame(frame);
  }

  requestAnimationFrame(frame);
});
```

**Implementation notes per decisions:**
- D-19: Uses `requestAnimationFrame` on `DOMContentLoaded` — pure JS, no CSS animation
- D-20: Duration is 800ms with quadratic ease-out (`1 - (1-t)^2`)
- D-21: Grade letter cycles through actual grade boundaries (F, D, C, B, A) as score counts up. Note: there is no grade 'E' in the grading system (confirmed in charts.py `_score_to_grade`). The cycle follows actual grade thresholds: F (<50), D (50-64), C (65-79), B (80-89), A (90+)
- D-22: Fires on every page load (not gated by sessionStorage)
- D-23: Target element is `.score-overall-badge`, reads `data-score` for the numeric final value
- The `finalGrade` is preserved from the element's initial text content and snapped to at the end (progress === 1), ensuring the displayed grade always matches the server-computed grade
